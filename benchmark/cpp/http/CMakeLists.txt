function(galay_add_http_benchmark target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::http)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    if(GALAY_HTTP_COROUTINE_WORKAROUND_ENABLED)
        target_compile_options(${target_name} PRIVATE ${GALAY_HTTP_COROUTINE_WORKAROUND_FLAGS})
    endif()
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON CXX_EXTENSIONS OFF)
endfunction()

galay_add_http_benchmark(benchmark_http_echo_server_throughput b1_echo_server_throughput.cc)
galay_add_http_benchmark(benchmark_http_client_throughput b2_client_throughput.cc)
galay_add_http_benchmark(benchmark_http_https_echo_throughput b14_https_echo_throughput.cc)
galay_add_http_benchmark(benchmark_http_header_parser_throughput b15_header_parser_throughput.cc)
