function(galay_add_tracing_benchmark target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::tracing)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

set(GALAY_TRACING_BENCHMARK_SCENARIO_b1_disabled_log disabled_log_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b2_enabled_log enabled_log_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b3_span_scope span_scope_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b4_traceparent traceparent_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b5_otlp_json_exporter otlp_json_exporter_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b6_structured_event structured_event_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b7_batch_processor_schedule batch_processor_schedule_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b8_moodycamel_batch_queue moodycamel_batch_queue_throughput)
set(GALAY_TRACING_BENCHMARK_SCENARIO_b9_http_header_integration http_header_integration_throughput)

file(GLOB GALAY_TRACING_BENCHMARK_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/b*.cc")
foreach(benchmark_source IN LISTS GALAY_TRACING_BENCHMARK_SOURCES)
    get_filename_component(benchmark_name "${benchmark_source}" NAME_WE)
    string(REGEX REPLACE "^b[0-9]+_" "" scenario_name "${benchmark_name}")
    set(scenario_override "GALAY_TRACING_BENCHMARK_SCENARIO_${benchmark_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    set(target_name "benchmark_tracing_${scenario_name}")

    if(benchmark_name STREQUAL "b9_http_header_integration" AND NOT TARGET galay::http)
        continue()
    endif()

    if(benchmark_name STREQUAL "b8_moodycamel_batch_queue")
        add_executable(${target_name} ${benchmark_source})
        target_include_directories(${target_name} SYSTEM PRIVATE "${GALAY_CONCURRENTQUEUE_INCLUDE_DIR}")
        set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
    else()
        galay_add_tracing_benchmark(${target_name} ${benchmark_source})
        if(benchmark_name STREQUAL "b9_http_header_integration")
            target_link_libraries(${target_name} PRIVATE galay::http)
        endif()
    endif()
endforeach()
