function(galay_mongo_configure_example_target target_name)
    set_target_properties(${target_name} PROPERTIES
        CXX_STANDARD 23
        CXX_STANDARD_REQUIRED ON
        CXX_EXTENSIONS OFF
    )
endfunction()

function(galay_add_mongo_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::mongo)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include ${CMAKE_CURRENT_SOURCE_DIR})
    galay_mongo_configure_example_target(${target_name})
endfunction()

function(galay_add_mongo_import_example target_name source_file)
    if(NOT GALAY_MONGO_CPP23_MODULES_EFFECTIVE)
        return()
    endif()
    galay_add_mongo_example(${target_name} ${source_file})
    if(GALAY_MONGO_IMPORT_COMPILE_OPTIONS)
        target_compile_options(${target_name} PRIVATE ${GALAY_MONGO_IMPORT_COMPILE_OPTIONS})
    endif()
endfunction()

set(GALAY_MONGO_EXAMPLE_SCENARIO_e1_ping sync_ping)
set(GALAY_MONGO_EXAMPLE_SCENARIO_e2_ping async_ping)

file(GLOB GALAY_MONGO_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_MONGO_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    string(REGEX REPLACE "^e[0-9]+_" "" scenario_name "${example_name}")
    set(scenario_override "GALAY_MONGO_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_mongo_example(example_mongo_include_${scenario_name} ${example_source})
endforeach()

    endforeach()
