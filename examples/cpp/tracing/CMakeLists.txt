function(galay_add_tracing_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::tracing)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

function(galay_add_tracing_import_example target_name source_file)
    if(NOT TARGET galay-tracing::galay-tracing-modules)
        return()
    endif()
    galay_add_tracing_example(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay-tracing::galay-tracing-modules)
    set_target_properties(${target_name} PROPERTIES CXX_SCAN_FOR_MODULES ON)
endfunction()

file(GLOB GALAY_TRACING_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_TRACING_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    string(REGEX REPLACE "^e[0-9]+_" "" scenario_name "${example_name}")
    galay_add_tracing_example(example_tracing_include_${scenario_name} ${example_source})
endforeach()

file(GLOB GALAY_TRACING_IMPORT_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/import/*.cc")
foreach(example_source IN LISTS GALAY_TRACING_IMPORT_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    string(REGEX REPLACE "^e[0-9]+_" "" scenario_name "${example_name}")
    galay_add_tracing_import_example(example_tracing_import_${scenario_name} ${example_source})
endforeach()
