function(galay_add_http_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::http)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include ${CMAKE_CURRENT_SOURCE_DIR})
    if(GALAY_HTTP_COROUTINE_WORKAROUND_ENABLED)
        target_compile_options(${target_name} PRIVATE ${GALAY_HTTP_COROUTINE_WORKAROUND_FLAGS})
    endif()
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

function(galay_add_http_import_example target_name source_file)
    if(NOT GALAY_HTTP_CPP23_MODULES_EFFECTIVE)
        return()
    endif()
    galay_add_http_example(${target_name} ${source_file})
    set_target_properties(${target_name} PROPERTIES CXX_EXTENSIONS OFF CXX_SCAN_FOR_MODULES ON)
endfunction()

file(GLOB GALAY_HTTP_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_HTTP_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    if(example_name MATCHES "https" AND NOT GALAY_SSL_FEATURE_ENABLED)
        continue()
    endif()
    string(REGEX REPLACE "^e[0-9]+_" "" scenario_name "${example_name}")
    galay_add_http_example(example_http_include_${scenario_name} ${example_source})
endforeach()

file(GLOB GALAY_HTTP_IMPORT_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/import/*.cc")
foreach(example_source IN LISTS GALAY_HTTP_IMPORT_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    if(example_name MATCHES "https" AND NOT GALAY_SSL_FEATURE_ENABLED)
        continue()
    endif()
    string(REGEX REPLACE "^e[0-9]+_" "" scenario_name "${example_name}")
    galay_add_http_import_example(example_http_import_${scenario_name} ${example_source})
endforeach()

file(GLOB GALAY_HTTP_MANUAL_EXAMPLE_SOURCES CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/include/manual/*.cc"
)
foreach(example_source IN LISTS GALAY_HTTP_MANUAL_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    if(example_name MATCHES "https" AND NOT GALAY_SSL_FEATURE_ENABLED)
        continue()
    endif()

    set(example_target "example_http_manual_${example_name}")
    galay_add_http_example(${example_target} ${example_source})
    target_include_directories(${example_target} PRIVATE ${PROJECT_SOURCE_DIR})
    file(READ "${example_source}" example_source_content)
    if(example_source_content MATCHES "#include[^\n]+[<\"]galay/cpp/galay-http2/")
        target_link_libraries(${example_target} PRIVATE galay::http2)
    endif()
    if(example_source_content MATCHES "#include[^\n]+[<\"]galay/cpp/galay-ws/")
        target_link_libraries(${example_target} PRIVATE galay::ws)
    endif()
endforeach()
