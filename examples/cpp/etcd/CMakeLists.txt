function(galay_etcd_configure_example_target target_name)
    set_target_properties(${target_name} PROPERTIES
        CXX_STANDARD 23
        CXX_STANDARD_REQUIRED ON
        CXX_EXTENSIONS OFF
    )
endfunction()

function(galay_add_etcd_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::etcd)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    set_property(TARGET ${target_name} PROPERTY NO_SYSTEM_FROM_IMPORTED ON)
    galay_etcd_configure_example_target(${target_name})
endfunction()

function(galay_add_etcd_import_example target_name source_file)
    if(NOT TARGET galay-etcd-modules)
        return()
    endif()
    galay_add_etcd_example(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay-etcd-modules)
    set_target_properties(${target_name} PROPERTIES CXX_SCAN_FOR_MODULES ON)
endfunction()

set(GALAY_ETCD_EXAMPLE_SCENARIO_e1_basic sync_basic)
set(GALAY_ETCD_EXAMPLE_SCENARIO_e2_basic async_basic)

file(GLOB GALAY_ETCD_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_ETCD_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    string(REGEX REPLACE "^e[0-9]+_" "" scenario_name "${example_name}")
    set(scenario_override "GALAY_ETCD_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_etcd_example(example_etcd_include_${scenario_name} ${example_source})
endforeach()
