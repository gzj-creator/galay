function(galay_add_ssl_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::ssl)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

function(galay_add_ssl_import_example target_name source_file)
    if(NOT TARGET galay-ssl-modules)
        return()
    endif()
    galay_add_ssl_example(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay-ssl-modules)
    set_target_properties(${target_name} PROPERTIES CXX_SCAN_FOR_MODULES ON)
endfunction()

set(GALAY_SSL_EXAMPLE_SCENARIO_e1_echo echo_server)
set(GALAY_SSL_EXAMPLE_SCENARIO_e2_echo echo_client)
set(GALAY_SSL_EXAMPLE_SCENARIO_e3_custom custom_context)
set(GALAY_SSL_EXAMPLE_SCENARIO_e4_builder builder_config)

file(GLOB GALAY_SSL_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_SSL_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_SSL_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_ssl_example(example_ssl_include_${scenario_name} ${example_source})
endforeach()

file(GLOB GALAY_SSL_IMPORT_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/import/*.cc")
foreach(example_source IN LISTS GALAY_SSL_IMPORT_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_SSL_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_ssl_import_example(example_ssl_import_${scenario_name} ${example_source})
endforeach()
