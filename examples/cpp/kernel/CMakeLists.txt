function(galay_add_kernel_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::kernel)
    target_include_directories(${target_name}
        PRIVATE
            ${PROJECT_SOURCE_DIR}
            ${CMAKE_BINARY_DIR}/include
    )
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

function(galay_add_kernel_import_example target_name source_file)
    if(NOT TARGET galay-kernel-modules)
        return()
    endif()
    galay_add_kernel_example(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay-kernel-modules)
    set_target_properties(${target_name} PROPERTIES CXX_SCAN_FOR_MODULES ON)
endfunction()

file(GLOB GALAY_KERNEL_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_KERNEL_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_KERNEL_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_kernel_example(example_kernel_include_${scenario_name} ${example_source})
endforeach()

file(GLOB GALAY_KERNEL_IMPORT_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/import/*.cc")
foreach(example_source IN LISTS GALAY_KERNEL_IMPORT_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_KERNEL_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_kernel_import_example(example_kernel_import_${scenario_name} ${example_source})
endforeach()
