function(galay_add_rpc_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::rpc)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

function(galay_add_rpc_import_example target_name source_file)
    if(NOT TARGET galay-rpc-modules)
        return()
    endif()
    galay_add_rpc_example(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay-rpc-modules)
    set_target_properties(${target_name} PROPERTIES CXX_SCAN_FOR_MODULES ON)
endfunction()

set(GALAY_RPC_EXAMPLE_SCENARIO_e1_echo echo_server)
set(GALAY_RPC_EXAMPLE_SCENARIO_e2_echo echo_client)
set(GALAY_RPC_EXAMPLE_SCENARIO_e3_stream stream_server)
set(GALAY_RPC_EXAMPLE_SCENARIO_e4_stream stream_client)
set(GALAY_RPC_EXAMPLE_SCENARIO_e5_managed_client managed_client)
set(GALAY_RPC_EXAMPLE_SCENARIO_e6_etcd_discovery etcd_discovery)
set(GALAY_RPC_EXAMPLE_SCENARIO_e7_tls_echo tls_echo)

file(GLOB GALAY_RPC_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_RPC_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_RPC_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_rpc_example(example_rpc_include_${scenario_name} ${example_source})
endforeach()

file(GLOB GALAY_RPC_IMPORT_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/import/*.cc")
foreach(example_source IN LISTS GALAY_RPC_IMPORT_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_RPC_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    if(example_name STREQUAL "e6_etcd_discovery" AND NOT TARGET galay-rpc-etcd-modules)
        continue()
    endif()
    set(example_target "example_rpc_import_${scenario_name}")
    galay_add_rpc_import_example(${example_target} ${example_source})
    if(example_name STREQUAL "e6_etcd_discovery")
        target_link_libraries(${example_target} PRIVATE galay-rpc-etcd-modules)
    endif()
endforeach()

file(GLOB GALAY_RPC_MANUAL_EXAMPLE_SOURCES CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/include/manual/*.cc"
