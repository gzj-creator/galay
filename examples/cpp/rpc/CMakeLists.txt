function(galay_add_rpc_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::rpc)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include)
    set_target_properties(${target_name} PROPERTIES CXX_STANDARD 23 CXX_STANDARD_REQUIRED ON)
endfunction()

function(galay_add_rpc_import_example target_name source_file)
    if(NOT TARGET galay-rpc-modules)
        return()
    endif()
    galay_add_rpc_example(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay-rpc-modules)
    set_target_properties(${target_name} PROPERTIES CXX_SCAN_FOR_MODULES ON)
endfunction()
