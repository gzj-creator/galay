function(galay_add_mysql_example target_name source_file)
    add_executable(${target_name} ${source_file})
    target_link_libraries(${target_name} PRIVATE galay::mysql)
    target_include_directories(${target_name} PRIVATE ${CMAKE_BINARY_DIR}/include ${CMAKE_CURRENT_SOURCE_DIR})
    set_target_properties(${target_name} PROPERTIES
        CXX_STANDARD 23
        CXX_STANDARD_REQUIRED ON
        CXX_EXTENSIONS OFF
    )
endfunction()

function(galay_add_mysql_import_example target_name source_file)
    if(NOT GALAY_MYSQL_CPP23_MODULES_EFFECTIVE)
        return()
    endif()
    galay_add_mysql_example(${target_name} ${source_file})
endfunction()

set(GALAY_MYSQL_EXAMPLE_SCENARIO_e1_query sync_query)
set(GALAY_MYSQL_EXAMPLE_SCENARIO_e2_query async_query)
set(GALAY_MYSQL_EXAMPLE_SCENARIO_e3_pool pool)
set(GALAY_MYSQL_EXAMPLE_SCENARIO_e4_prepared prepared_statement)
set(GALAY_MYSQL_EXAMPLE_SCENARIO_e5_pipeline pipeline)

file(GLOB GALAY_MYSQL_INCLUDE_EXAMPLE_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc")
foreach(example_source IN LISTS GALAY_MYSQL_INCLUDE_EXAMPLE_SOURCES)
    get_filename_component(example_name "${example_source}" NAME_WE)
    set(scenario_name "${example_name}")
    set(scenario_override "GALAY_MYSQL_EXAMPLE_SCENARIO_${example_name}")
    if(DEFINED "${scenario_override}")
        set(scenario_name "${${scenario_override}}")
    endif()
    galay_add_mysql_example(example_mysql_include_${scenario_name} ${example_source})
endforeach()
